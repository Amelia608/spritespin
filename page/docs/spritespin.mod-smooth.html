    (function ($, SpriteSpin) {
  "use strict";

  function loop(data, smooth) {
    smooth.timeout = window.setTimeout(function(){
      tick(data, smooth);
    }, smooth.ms);
  }

  function tick(data, smooth) {
    if (!smooth.canvas) {
      return;
    }

    var animation, i, context = smooth.canvas[0].getContext("2d");
    var toRemove = [];
    context.clearRect(0, 0, data.width, data.height);
    for (i = 0; i < smooth.animations.length; i += 1) {

      animation = smooth.animations[i];
      animation.t -= smooth.ms;
      if (animation.t < 0) {
        animation.t = 0;
        toRemove.push(animation);
      }

      var index = data.lane * data.frames + animation.frame;
      var img = data.images[index];
      
      if (img && img.complete !== false){
        context.globalAlpha = Math.max(0, animation.t / smooth.t - 0.25);
        context.drawImage(img, 0, 0, data.width, data.height);
      }
    }
    
    for (i = 0; i < toRemove.length; i += 1) {
      var index = $.inArray(toRemove[i], smooth.animations);
      if (index >= 0) {
        smooth.animations.splice(index, 1);
      }
    }

    loop(data, smooth);
  }

  SpriteSpin.registerModule('smooth', {

    onLoad: function(e, data){
      data.modSmooth = data.modSmooth || {
        canvas: $("<canvas class='smooth-layer'></canvas>"),
        animations: [],
        ms: 20,
        t: 140
      }
      var smooth = data.modSmooth;
      var css = SpriteSpin.calculateInnerLayout(data);
      smooth.canvas[0].width = data.width;
      smooth.canvas[0].height = data.height;
      smooth.canvas.css(css).show();
      data.target.append(smooth.canvas);
      if (smooth.timeout) {
        window.clearTimeout(smooth.timeout)
      }
      smooth.timeout = loop(data, smooth);
    },

    onFrameChanged: function(e, data){
      var d = Math.abs(data.frame - data.lastFrame);
      d = Math.min((d / 2)|0, 3);
      data.modSmooth.canvas.css({
        '-webkit-filter': 'blur(' + d + 'px)',
        'filter': 'blur(' + d + 'px)'
      });
      data.modSmooth.animations.unshift({
        frame: data.frame,
        lane: data.lane,
        t: data.modSmooth.t
      });

    }
  });

}(window.jQuery || window.Zepto || window.$, window.SpriteSpin));
